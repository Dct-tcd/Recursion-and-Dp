class Solution{
public:
  int dp[110][110];
  
   int help(int arr[] , int i , int j)
   {
       if(i == j )
          return 0;
          
          if(dp[i][j] != -1)
             return dp[i][j];
          
          int ans = INT_MAX;
          
          for(int k = i; k < j ; k++)
          {
              int t = help(arr, i , k) ;
              int t3 =help( arr, k+1 , j) ;
             
              dp[i][k] = t;
              dp[k+1][j] = t3;
              
              int t1 = arr[i-1]*arr[k]*arr[j];
              
              t += t1;
              t += t3;
              
              ans = min(ans, t);
            
          }
          return  dp[i][j] = ans;
       
   }
   
    int matrixMultiplication(int N, int arr[])
    {
        // code here
        memset(dp,-1,sizeof(dp));
        return help(arr,1,N-1);
    }
};

            cin>>arr[i];
